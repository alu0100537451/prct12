  def traspuesta#Método que calcula la traspuesta de una matriz
          i=0
          mtrasp = Array.new(@filas) {Array.new(self.columnas)}
          while i < @filas
                  j=0
                  while j < @columnas
                          mtrasp[i][j] = self.matriz[j][i]
                          j+=1
                  end
                  i+=1
          end
          Matrices.new(mtrasp)
  end

  def opuesta#Método que calcula el opuesto de una matriz
